33 Single Command/Response packet:
34 +----------+-----------+-----------+------------+-----------+
35 | 16 bits | 4 bits | 12 bits | 32 bits | 16 bits |
36 | Preamble | Command | Address | Data | CRC16 |
37 +----------+-----------+-----------+------------+-----------+
38

40
41 +----------+-----------+---------------+---------------+

44 +----------+-----------+---------------+---------------+
45 +-----------------+------------+-----------+
46 % 16 bits | Variable | 16 bits |
47 % Payload Length | Payload | CRC16 |
48 +-----------------+------------+-----------+
49

108 def calculateCRC(self):
109 """
110 Calculate CRC-CCIT (XModem) (2 bytes) from 8 byte packet for send and rcv
111
112 Returns:
113 CRC as hexadecimal string without '0x'
114 """
115 preamble = self.preamble
116 crc = self.crc
117 self.crc = ""
118 self.preamble = ""
119
120 CRC_dec = crc16pure.crc16xmodem(str2bytes(self.pktStr()))
121 # input = int type decimal, output = hex string with 0x at the beginning
122 CRC_hex_0x = "0x%0.4X" % CRC_dec
123 # make all hex letters lower case for comparison
124 CRC_hex = CRC_hex_0x.lower()
125 # input = hex string with 0x at the beginning, output = hex str with 0x removed
126 CRC_hex = CRC_hex[2:]
127 self.preamble = preamble
128 self.crc = crc
129 return CRC_hex

137 def checkReadPacket(self, resppkt):
138 """
139 Confirm that Read Single occurred without error
140 Args:
141 resppkt: response packet
142
143 Returns:
144 tuple (error string, response packet as string)
145 """
146 err = ""
147 if int(resppkt.cmd.upper(), 16) - int(self.cmd.upper(), 16) != 0x8:
148 err = "invalid command; "
149 if resppkt.addr.upper() != self.addr.upper():
