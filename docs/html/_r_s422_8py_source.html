30class RS422:
31 """
32 Code to manage RS422 connection. Will automatically query available COM interfaces
33 until a board is found. Use the 'port=x' parameter in cameraAssembler call to
34 specify a particular COM interface.
35
36 Exposed methods:
37 arm() - Puts camera into wait state for external trigger
38 readFrames() - waits for data ready register flag, then copies camera image data
39 into numpy arrays
40 readoff() - waits for data ready register flag, then copies camera image data
41 into numpy arrays; returns payload, payload size, and error message
42 sendCMD(pkt) - sends packet object via serial port
43 readSerial(size, timeout) - read 'size' bytes from serial port
44 writeSerial(cmd) - submits string 'cmd' (assumes string is preformed packet)
45 closeDevice() - close serial connections
46 """

170 def sendCMD(self, pkt):
171 """
172 Submit packet and verify response packet. Recognizes readoff packet and adjusts
173 read size and timeout appropriately
174
175 Args:
176 pkt: Packet object
177
178 Returns:
179 tuple (error, response string)
180 """
181 pktStr = pkt.pktStr()
182 logging.debug(self.logdebug + "sendCMD packet: " + str(pktStr))
183 self._ser.flushInput()
184 time.sleep(0.01) # wait 10 ms in between flushing input and output buffers
185 self._ser.flushOutput()
186 self.ca.writeSerial(pktStr)
187 err0 = ""
188 err = ""
189 resp = ""
190 tries = 3 # TODO: make a function parameter?
191
192 if (
193 hasattr(self.ca, "board")
194 and pktStr[4] == "0"
195 and pktStr[5:8] == self.ca.board.registers["SRAM_CTL"]
196 ):
197 # download data payload
198 logging.info(
199 self.loginfo + "Payload size (bytes) = " + str(self.payloadsize)
200 )
201 crcresp0 = ""
202 crcresp1 = ""
203 smallresp = ""
204 emptyResponse = False
205 wrongSize = False
206 # TODO: refactor payload error management to another method
207 for i in range(tries):
208 err, resp = self.readSerial(
209 self.payloadsize + 20, timeout=self.datatimeout
210 )
211 if err:
212 logging.error(
213 self.logerr + "sendCMD: read payload failed " + pktStr + err
214 )
215 self.ca.payloaderror = True
216 else:
217 if not len(resp):
218 err0 = self.logerr + "sendCMD: empty response from board"
219 logging.error(err0)
220 emptyResponse = True
221 self.ca.payloaderror = True
222 elif len(resp) != 2 * (self.payloadsize + 20):
223 err0 = (
224 self.logerr
225 + "sendCMD: incorrect response; expected "
226 + str(self.payloadsize + 20)
227 + " bytes, received "
228 + str(len(resp) // 2)
229 )
230 logging.error(err0)
231 wrongSize = True
232 smallresp = resp
233 self.ca.payloaderror = True
234 elif not checkCRC(resp[4:20]):
235 err0 = (
236 self.logerr
237 + "sendCMD: "
238 + pktStr
239 + " - payload preface CRC fail"
240 )
241 logging.error(err0)
242 self.ca.payloaderror = True
243 crcresp1 = resp
244 elif not checkCRC(resp[24:]):
245 err0 = (
246 self.logerr + "sendCMD: " + pktStr + " - payload CRC fail"
247 )
248 logging.error(err0)
249 self.ca.payloaderror = True
250 crcresp0 = resp
251 err += err0
252 time.sleep(5)
253 if self.ca.payloaderror:
254 # keep best results over multiple tries; e.g., if first try is
255 # bad CRC and second try is an incomplete payload, use the
256 # first payload
257 if i == tries - 1:
258 if crcresp0:
259 logging.error(
260 self.logerr + "sendCMD: Unable to acquire "
261 "CRC-confirmed payload after "
262 + str(tries)
263 + " attempts. Continuing with unconfirmed payload"
264 )
265 resp = crcresp0
266 elif crcresp1:
267 logging.error(
268 self.logerr + "sendCMD: Unable to acquire "
269 "CRC-confirmed readoff after "
270 + str(tries)
271 + " attempts. Continuing with unconfirmed payload"
272 )
273 resp = crcresp1
274 elif wrongSize:
275 logging.error(
276 self.logerr + "sendCMD: Unable to acquire complete "
277 "payload after "
278 + str(tries)
279 + " attempts. Dumping datastream to file."
280 )
281 resp = smallresp
282 self.ca.dumpNumpy(resp)
283 elif emptyResponse:
284 logging.error(
285 self.logerr + "sendCMD: Unable to acquire any "
286 "payload after " + str(tries) + " attempts."
287 )
288 else:
289 logging.info(
290 self.loginfo + "Retrying download, attempt #" + str(i + 1)
291 )
292 err = ""
293 err0 = ""
294 self.ca.payloaderror = False
295 self.ca.writeSerial(pktStr)
296 else:
297 logging.info(self.loginfo + "Download successful")
298 if self.ca.boardname == "llnl_v4":
299 # self.ca.setSubregister('SWACK','1')
300 pass
301 break
302
303 else:
304 # non-payload messages and workaround for initial setup before board object
305 # has been initialized
306 time.sleep(0.03)
307 self._ser.timeout = 0.02
308 err, resp = self.readSerial(10)
309 logging.debug(self.logdebug + "sendCMD response: " + str(resp))
310 if err:
311 logging.error(
312 self.logerr + "sendCMD: readSerial failed (regular packet) " + err
313 )
314 elif not checkCRC(resp[4:20]):
315 err = self.logerr + "sendCMD- regular packet CRC fail: " + resp
316 logging.error(err)
317 return err, resp

348 Copies image data from board into numpy arrays.
349

353 timeout: passed to waitForSRAM; after this many seconds begin copying data

355 WARNING: If acquisition fails, the SRAM will not contain a current image,

359 columns: 1 for single image per frame, 2 for separate hemisphere images

361 Returns:

363
364 """
365 frames, _, _ = self.readoff(waitOnSRAM, timeout, fast, columns)

368 def readoff(self, waitOnSRAM, timeout, fast, columns=1):
369 """
370 Copies image data from board into numpy arrays; returns data, length of data,
371 and error messages. Use 'readFrames()' unless you require this additional
372 information
373
374 Args:
375 waitOnSRAM: if True, wait until SRAM_READY flag is asserted to begin copying
376 data
377 timeout: passed to waitForSRAM; after this many seconds begin copying data
378 irrespective of SRAM_READY status; 'zero' means wait indefinitely
379 WARNING: If acquisition fails, the SRAM will not contain a current image,
380 but the code will copy the data anyway
381 fast: if False, parse and convert frames to numpy arrays; if True, return
382 unprocessed text stream
383 columns: 1 for single image per frame, 2 for separate hemisphere images
384
385 Returns:
386 tuple (list of numpy arrays OR raw text stream, length of downloaded payload
387 in bytes, payload error flag)
388 NOTE: This reduces readoff by <1 second, so will have no noticeable impact
389 when using RS422
390 """
391 logging.info(self.loginfo + "readoff")
392 logging.debug(
393 self.logdebug
394 + "readoff: waitonSRAM = "
395 + str(waitOnSRAM)
396 + "; timeout = "
397 + str(timeout)
398 + "; fast = "
399 + str(fast)
400 )
401 errortemp = False
402
403 # Wait for data to be ready on board, turns off error messaging
404 # Skip wait only if explicitly tagged 'False' ('None' defaults to True)
405 if waitOnSRAM is not False:
406 logging.getLogger().setLevel(logging.CRITICAL)
407 self.ca.waitForSRAM(timeout)
408 logging.getLogger().setLevel(self.ca.verblevel)
409
410 # Retrieve data
411 err, rval = self.ca.readSRAM()
412 if err:
413 logging.error(self.logerr + "Error detected in readSRAM")
414 time.sleep(0.3)
415 logging.debug(self.logdebug + "readoff: first 64 chars: " + str(rval[0:64]))
416 # extract only the read burst data. Remove header & CRC footer
417 read_burst_data = rval[36:-4]
418
419 # Payload size as string implied by provided parameters
420 expectedlength = (
421 4
422 * (self.ca.sensor.lastframe - self.ca.sensor.firstframe + 1)
423 * (self.ca.sensor.lastrow - self.ca.sensor.firstrow + 1)
424 * self.ca.sensor.width
425 )
426 padding = expectedlength - len(read_burst_data)
427 if padding:
428 logging.warning(
429 "{logwarn}readoff: Payload is shorter than expected."
430 " Padding with '0's".format(logwarn=self.logwarn)
431 )
432 read_burst_data = read_burst_data.ljust(expectedlength, "0")
433
434 if fast:
435 return read_burst_data, len(read_burst_data) // 2, errortemp
436 else:
437 parsed = generateFrames(self.ca, read_burst_data, columns)
438 return parsed, len(read_burst_data) // 2, errortemp
